    # ─── Trạng thái ───────────────────────────────
    STATUS_CHOICES = [

    class Meta:

class Episode(models.Model):

                    Movie, on_delete=models.CASCADE,
                    related_name='episodes')

3.1 — Giải Thích Chi Tiết Lệnh FFmpeg
▼
Shell — Lệnh FFmpeg đầy đủ với giải thích
ffmpeg \
  -i input.mp4 \                  # File đầu vào
  -c:v libx264 \                  ★ MỚI: chỉ định codec video H.264
  -profile:v baseline \          # Profile H.264 tương thích cao nhất
  -level 3.0 \                   # Level 3.0 = max 720p@30fps — đủ cho LAN
  -c:a aac \                     ★ MỚI: codec audio AAC (tương thích HLS)
  -b:a 128k \                    ★ MỚI: bitrate audio 128kbps
  -movflags +faststart \        ★ MỚI: cho phép stream trước khi tải xong
  -start_number 0 \             # Segment đầu tiên đánh số từ 0
  -hls_time 10 \                # Mỗi segment = 10 giây video
  -hls_list_size 0 \            # 0 = giữ tất cả segments trong .m3u8
  -hls_flags single_file \      # Byte-Range: gộp tất cả vào 1 file .ts
  -f hls \                      # Format output: HLS
  output.m3u8                   # Tạo ra: output.m3u8 + output.ts
Byte-Range HLS (single_file) là gì? Thay vì tạo hàng trăm file .ts nhỏ (output0.ts, output1.ts...), FFmpeg tạo ra MỘT file output.ts lớn và file output.m3u8 chứa "map" offset byte của từng segment. Khi trình duyệt cần đoạn từ giây 30-40, nó chỉ tải byte 4MB đến 8MB của file .ts, không tải toàn bộ. Kết quả: ít file hơn, dễ quản lý hơn, phù hợp với mạng LAN tốc độ cao.
Thông sốÝ nghĩaThay đổi nếu...
-hls_time 10Mỗi chunk = 10 giâyMạng chậm → tăng lên 30s
-profile baselineTương thích maxChỉ dùng Chrome → đổi main/high
-b:a 128kAudio bitrateMuốn chất lượng âm thanh cao → 192k

video_processor/processor.py
import os, sys, subprocess, shutil, logging
from pathlib import Path
from datetime import datetime

# ─── Thiết lập để dùng Django ORM ─────────────────
sys.path.insert(0, str(Path(__file__).resolve().parent.parent))
os.environ.setdefault('DJANGO_SETTINGS_MODULE', 'core.settings')

import django
django.setup()

from movies.models import Movie, Episode

# ─── Logging ──────────────────────────────────────
logging.basicConfig(
    filename='logs/processor.log',
    level=logging.INFO,
    format='%(asctime)s [%(levelname)s] %(message)s',
    datefmt='%Y-%m-%d %H:%M:%S'
)
log = logging.getLogger(__name__)

# ─── Đường dẫn ────────────────────────────────────
BASE_DIR   = Path(__file__).resolve().parent.parent
RAW_DIR    = Path(__file__).parent / 'raw_videos'
MEDIA_DIR  = BASE_DIR / 'media'
HLS_DIR    = MEDIA_DIR / 'hls_streams'
DONE_DIR   = Path(__file__).parent / 'processed'   # Lưu file đã xử lý

# Tạo thư mục nếu chưa có
for d in [RAW_DIR, HLS_DIR, DONE_DIR]:
    d.mkdir(parents=True, exist_ok=True)


def get_video_duration(input_path: Path) -> int:
    """Lấy thời lượng video (giây) bằng ffprobe."""
    result = subprocess.run([
        'ffprobe', '-v', 'quiet',
        '-print_format', 'csv=p=0',
        '-show_entries', 'format=duration',
        str(input_path)
    ], capture_output=True, text=True)
    try:
        return int(float(result.stdout.strip()))
    except (ValueError, AttributeError):
        return 0


def encode_hls(input_path: Path, output_dir: Path) -> bool:
    """Chạy FFmpeg encode HLS. Trả về True nếu thành công."""
    output_dir.mkdir(parents=True, exist_ok=True)
    output_m3u8 = output_dir / 'output.m3u8'

    cmd = [
        'ffmpeg', '-i', str(input_path),
        '-c:v', 'libx264',
        '-profile:v', 'baseline',
        '-level', '3.0',
        '-c:a', 'aac',
        '-b:a', '128k',
        '-start_number', '0',
        '-hls_time', '10',
        '-hls_list_size', '0',
        '-hls_flags', 'single_file',
        '-f', 'hls',
        str(output_m3u8),
    ]
    try:
        log.info(f"Bắt đầu encode: {input_path.name}")
        result = subprocess.run(
            cmd,
            capture_output=True, text=True,
            timeout=3600        # Timeout 1 tiếng — phim dài ~2GB
        )
        if result.returncode != 0:
            log.error(f"FFmpeg lỗi: {result.stderr[-500:]}")
            return False
        log.info(f"Encode thành công: {input_path.name}")
        return True
    except subprocess.TimeoutExpired:
        log.error(f"Timeout khi encode {input_path.name}")
        return False
    except Exception as e:
        log.error(f"Lỗi không xác định: {e}")
        return False


def parse_filename(filename: str):
    """
    Phân tích tên file theo quy ước:
    ten-phim_tap-1.mp4  → movie_slug='ten-phim', ep=1
    ten-phim.mp4        → movie_slug='ten-phim', ep=1
    """
    stem = Path(filename).stem
    if '_tap-' in stem:
        parts = stem.split('_tap-')
        return parts[0], int(parts[1])
    return stem, 1


def process_video(mp4_path: Path):
    """Xử lý một file .mp4: encode + tạo DB record."""
    movie_slug, episode_num = parse_filename(mp4_path.name)

    # Tìm hoặc tạo Movie
    movie, created = Movie.objects.get_or_create(
        slug=movie_slug,
        defaults={'title': movie_slug.replace('-', ' ').title()}
    )
    if created:
        log.info(f"Tạo Movie mới: {movie.title}")

    # Bỏ qua nếu Episode đã tồn tại và ở trạng thái ready
    if Episode.objects.filter(
            movie=movie, episode_num=episode_num,
            status='ready').exists():
        log.info(f"Bỏ qua (đã tồn tại): {mp4_path.name}")
        return

    # Tạo Episode với trạng thái "processing"
    output_dir = HLS_DIR / movie_slug / f'tap-{episode_num}'
    video_url  = f'hls_streams/{movie_slug}/tap-{episode_num}/output.m3u8'

    episode, _ = Episode.objects.update_or_create(
        movie=movie, episode_num=episode_num,
        defaults={
            'video_url': video_url,
            'status'   : 'processing'
        }
    )

    # Lấy thời lượng trước khi encode
    duration = get_video_duration(mp4_path)

    # Encode
    if encode_hls(mp4_path, output_dir):
        episode.status   = 'ready'
        episode.duration = duration
        episode.save()

        # Di chuyển file gốc vào thư mục "processed"
        shutil.move(str(mp4_path), str(DONE_DIR / mp4_path.name))
        log.info(f"Hoàn thành: {mp4_path.name} → tap {episode_num}")
    else:
        episode.status = 'error'
        episode.save()
        log.error(f"Encode thất bại: {mp4_path.name}")


def main():
    mp4_files = list(RAW_DIR.glob('*.mp4'))
    if not mp4_files:
        log.info("Không có file .mp4 mới.")
        return

    log.info(f"Tìm thấy {len(mp4_files)} file .mp4 mới")
    for mp4 in sorted(mp4_files):
        process_video(mp4)


if __name__ == '__main__':
    main()
💡 Quy ước đặt tên file: Đặt file theo định dạng ten-phim_tap-3.mp4 để script tự nhận diện tên phim và số tập. Ví dụ: one-piece_tap-1.mp4, one-piece_tap-2.mp4.

JIKAN_BASE = "https://api.jikan.moe/v4"

# ★ QUAN TRỌNG: Rate limit — Jikan giới hạn 3 req/giây
RATE_DELAY = 0.5    # Chờ 0.5s giữa mỗi request


def jikan_search(title: str) -> dict | None:
    """Tìm kiếm anime theo tên. Trả về dict data hoặc None."""
    time.sleep(RATE_DELAY)          # Tránh bị block IP
    try:
        r = requests.get(
            f"{JIKAN_BASE}/anime",
            params={'q': title, 'limit': 1},
            timeout=10
        )
        r.raise_for_status()
        data = r.json().get('data', [])
        return data[0] if data else None
    except requests.RequestException as e:
        log.warning(f"Jikan API lỗi khi tìm '{title}': {e}")
        return None


def download_poster(movie: Movie, url: str) -> bool:
    """Tải ảnh poster và lưu vào ImageField."""
    if movie.poster:           # Đã có poster rồi, bỏ qua
        return False
    try:
        time.sleep(RATE_DELAY)
        r = requests.get(url, timeout=15)
        r.raise_for_status()
        ext  = url.rsplit('.', 1)[-1].lower() or 'jpg'
        fname = f"{movie.slug}.{ext}"
        movie.poster.save(fname, ContentFile(r.content), save=True)
        return True
    except Exception as e:
        log.warning(f"Tải poster lỗi: {e}")
        return False


def enrich_movie(movie: Movie):
    """Làm giàu thông tin cho một Movie từ Jikan."""
    log.info(f"Crawl metadata: {movie.title}")
    data = jikan_search(movie.title)
    if not data:
        log.warning(f"Không tìm thấy: {movie.title}")
        return

    # Cập nhật thông tin text
    movie.description  = data.get('synopsis',     '') or movie.description
    movie.title_en     = data.get('title_english','') or movie.title_en
    movie.release_year = (data.get('aired', {})
                              .get('prop', {})
                              .get('from', {})
                              .get('year'))
    movie.rating       = data.get('score')
    movie.save()

    # Xử lý genre (ManyToMany)
    for g in data.get('genres', []):
        genre, _ = Genre.objects.get_or_create(name=g['name'])
        movie.genres.add(genre)

    # Tải poster
    poster_url = (data.get('images', {})
                      .get('jpg', {})
                      .get('large_image_url'))
    if poster_url:
        download_poster(movie, poster_url)

    log.info(f"Hoàn thành: {movie.title} (rating={movie.rating})")


def main():
    # Chỉ crawl các movie chưa có description
    movies = Movie.objects.filter(description='')
    log.info(f"Crawl metadata cho {movies.count()} phim")
    for movie in movies:
        enrich_movie(movie)


if __name__ == '__main__':
    main()
⚠ Rate Limiting: Jikan API giới hạn 60 request/phút. Nếu crawl quá nhanh, IP của bạn sẽ bị block tạm thời. Đoạn time.sleep(RATE_DELAY) trước mỗi request là bắt buộc.

from django.shortcuts import render, get_object_or_404
from django.core.paginator import Paginator
from .models import Movie, Episode

def home(request):
    # Chỉ hiện phim visible, sắp xếp mới nhất
    movies = Movie.objects.filter(is_visible=True).prefetch_related('genres')

    # Tìm kiếm
    query = request.GET.get('q', '')
    if query:
        movies = movies.filter(title__icontains=query)

    # Phân trang — 24 phim/trang
    paginator = Paginator(movies, 24)
    page_obj  = paginator.get_page(request.GET.get('page'))

    return render(request, 'movies/home.html', {
        'page_obj': page_obj,
        'query'   : query,
    })


def movie_detail(request, slug):
    movie    = get_object_or_404(Movie, slug=slug, is_visible=True)
    episodes = movie.episodes.filter(status='ready')  # Chỉ tập sẵn sàng
    # Tập mặc định = tập đầu tiên
    ep_num   = request.GET.get('tap', 1)
    current_episode = episodes.filter(episode_num=ep_num).first() or episodes.first()

    return render(request, 'movies/detail.html', {
        'movie'          : movie,
        'episodes'       : episodes,
        'current_episode': current_episode,
    })

<script>
  const video   = document.getElementById('player');
  // Đường dẫn đến file .m3u8 — được render từ Django
  const m3u8Url = "/media/{{ current_episode.video_url }}";

  if (Hls.isSupported()) {
    // Chrome, Firefox, Edge — dùng hls.js
    const hls = new Hls({
      maxBufferLength: 60,          // Buffer tối đa 60 giây
      maxMaxBufferLength: 120,       // Tối đa 120 giây trong mạng LAN nhanh
    });
    hls.loadSource(m3u8Url);
    hls.attachMedia(video);
    hls.on(Hls.Events.MANIFEST_PARSED, () => video.play().catch(() => {}));
    hls.on(Hls.Events.ERROR, (_, data) => {
      if (data.fatal) console.error('HLS lỗi nghiêm trọng:', data);
    });
  } else if (video.canPlayType('application/vnd.apple.mpegurl')) {
    // Safari — native HLS support
    video.src = m3u8Url;
    video.play().catch(() => {});
  } else {
    video.insertAdjacentHTML('afterend',
      '<p>Trình duyệt không hỗ trợ HLS.</p>');
  }
</script>
hls.js vs video.js: hls.js là thư viện thuần HLS, nhỏ gọn hơn video.js (~150KB vs ~450KB). video.js đa năng hơn nhưng nặng hơn. Với yêu cầu đơn giản là phát HLS, hls.js là lựa chọn tối ưu. Safari đã hỗ trợ HLS native nên không cần thư viện.

6.2 — Caddyfile Hoàn Chỉnh với Headers Bảo Mật
▼
C:\Caddy\Caddyfile
# Thay 192.168.1.x bằng IP tĩnh của máy bạn
localhost:80, 192.168.1.x:80 {

    # ── Headers bảo mật ★ MỚI ────────────────────────
    header {
        X-Content-Type-Options  "nosniff"
        X-Frame-Options         "SAMEORIGIN"
        Referrer-Policy         "strict-origin-when-cross-origin"
        # Tắt server signature
        -Server
    }

    # ── Static files (CSS, JS, admin) ────────────────
    route /static/* {
        header Cache-Control "public, max-age=86400"   # Cache 1 ngày
        file_server {
            root C:\Projects\my_movie_web
        }
    }

    # ── Media files (HLS .m3u8, .ts, poster) ─────────
    route /media/* {
        # HLS manifest (.m3u8): không cache để luôn fresh
        @m3u8 path *.m3u8
        header @m3u8 Cache-Control "no-cache, no-store"

        # File .ts: cache mạnh vì nội dung không đổi
        @ts path *.ts
        header @ts Cache-Control "public, max-age=604800"  # 7 ngày

        file_server {
            root C:\Projects\my_movie_web
        }
    }

    # ── Django App (mọi request còn lại) ─────────────
    reverse_proxy localhost:8000 {
        # ★ MỚI: Header để Django nhận IP thật của client
        header_up X-Forwarded-For {remote_host}
        header_up X-Forwarded-Proto {scheme}
    }

    # ── Logging của Caddy ★ MỚI ──────────────────────
    log {
        output file C:\Caddy\access.log {
            roll_size 10mb
            roll_keep 3
        }
        format json
    }
}
Tại sao .m3u8 không cache nhưng .ts thì cache 7 ngày? File .m3u8 là "manifest" — nó chứa danh sách và offset của các segment. Nếu cache manifest mà sau đó bạn thay đổi video, trình duyệt vẫn đọc bản cũ. File .ts là nội dung video thực sự — một khi đã encode xong, nó không bao giờ thay đổi, nên cache dài là an toàn và tăng hiệu suất đáng kể cho lần xem lại.

@echo off

# Đăng ký Task Scheduler qua PowerShell (chạy với quyền Admin)
$action = New-ScheduledTaskAction -Execute "C:\Projects\start_django.bat"
$trigger = New-ScheduledTaskTrigger -AtStartup
$settings = New-ScheduledTaskSettingsSet -RestartCount 3 -RestartInterval (New-TimeSpan -Minutes 1)
Register-ScheduledTask -TaskName "MovieWeb-Django" -Action $action -Trigger $trigger -Settings $settings -RunLevel Highest

# Tương tự cho Caddy
$action2 = New-ScheduledTaskAction -Execute "C:\Caddy\caddy.exe" -Argument "run" -WorkingDirectory "C:\Caddy"
Register-ScheduledTask -TaskName "MovieWeb-Caddy" -Action $action2 -Trigger $trigger -Settings $settings -RunLevel Highest
Task Scheduler vs chạy tay: Task Scheduler đảm bảo cả Waitress và Caddy tự khởi động khi Windows bật, và tự restart nếu crash. -RestartCount 3 = thử restart 3 lần trước khi bỏ cuộc.
